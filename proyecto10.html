3. Programación Orientada a Objetos

Implementación de clases y herencia en Python

POO en Python
# Clase y herencia class Vehiculo: def __init__(self, marca, modelo): self.marca = marca self.modelo = modelo self.velocidad = 0 def acelerar(self, incremento): self.velocidad += incremento return f"{self.marca} {self.modelo} acelerando a {self.velocidad} km/h" class Coche(Vehiculo): def __init__(self, marca, modelo, puertas): super().__init__(marca, modelo) self.puertas = puertas def describir(self): return f"{self.marca} {self.modelo} con {self.puertas} puertas" # Crear instancias mi_coche = Coche("Toyota", "Corolla", 4) print(mi_coche.describir()) print(mi_coche.acelerar(50))
